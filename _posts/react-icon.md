## 安装

```bash
npm install react-icons
```

## 代码

### App.jsx

```jsx
import { SiHomebridge } from "react-icons/si";

//...
return (
  <div>
    <SiHomebridge />
  </div>
);
//...
```

### 设置大小

```jsx
import { SiHomebridge } from "react-icons/si";

//...
return (
  <div>
    <SiHomebridge size={200} />
  </div>
);
//...
```
